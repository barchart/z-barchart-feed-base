248            @Test
249            public void testCountPlaces() {
250    
251                    final SizeValue size = newSize(16);
252                    final PriceValue step = newPrice(25, -2);
253                    final UniBook book = new UniBook(COMBO, size, step);
254                    final UniBookRing asks = book.asks;
255                    final UniBookRing bids = book.bids;
256                    UniBookResult result;
257                    MarketDoBookEntry entry;
258    
259                    // ###
260    
261                    assertTrue(book.isEmpty(ASK));
262    
263                    assertEquals(0, asks.placeFromOffset(16 - 1));
264    
265                    entry = new DefBookEntry(MODIFY, ASK, DEFAULT, 1, newPrice(1000, -2),
266                                    newSize(13));
267                    book.make(entry);
268                    // System.out.println(book);
269    
270                    assertEquals(1, asks.placeFromOffset(16 - 1));
271    
272                    entry = new DefBookEntry(MODIFY, ASK, DEFAULT, 0, newPrice(1200, -2),
273                                    newSize(15));
274                    book.make(entry);
275                    // System.out.println(book);
276    
277                    assertEquals(2, asks.placeFromOffset(16 - 1));
278    
279                    entry = new DefBookEntry(MODIFY, ASK, DEFAULT, 0, newPrice(1100, -2),
280                                    newSize(17));
281                    book.make(entry);
282                    // System.out.println(book);
283    
284                    assertEquals(3, asks.placeFromOffset(16 - 1));
285                    assertEquals(1, asks.placeFromOffset(0));
286                    assertEquals(1, asks.placeFromOffset(3));
287                    assertEquals(2, asks.placeFromOffset(4));
288                    assertEquals(2, asks.placeFromOffset(6));
289                    assertEquals(3, asks.placeFromOffset(10));
290    
291                    entry = new DefBookEntry(MODIFY, ASK, DEFAULT, 1, newPrice(1050, -2),
292                                    newSize(19));
293                    book.make(entry);
294                    // System.out.println(book);
295                    // System.out.println(asks.count());
296    
297                    assertEquals(2, asks.placeFromOffset(4));
298                    assertEquals(3, asks.placeFromOffset(16 - 1));
299    
300                    entry = new DefBookEntry(MODIFY, ASK, DEFAULT, 0, newPrice(1425, -2),
301                                    newSize(21));
302                    book.make(entry);
303                    // System.out.println(book);
304    
305                    assertEquals(4, asks.placeFromOffset(16 - 1));
306    
307                    entry = new DefBookEntry(MODIFY, ASK, DEFAULT, 0, newPrice(1400, -2),
308                                    newSize(23));
309                    book.make(entry);
310                    // System.out.println(book);
311    
312                    assertEquals(4, asks.placeFromOffset(16 - 2));
313                    assertEquals(5, asks.placeFromOffset(16 - 1));
314    
315                    entry = new DefBookEntry(MODIFY, ASK, DEFAULT, 1, newPrice(1275, -2),
316                                    newSize(25));
317                    book.make(entry);
318                    // System.out.println(book);
319    
320                    assertEquals(1, asks.placeFromOffset(0));
321                    assertEquals(1, asks.placeFromOffset(2));
322                    assertEquals(2, asks.placeFromOffset(5));
323                    assertEquals(3, asks.placeFromOffset(6));
324                    assertEquals(3, asks.placeFromOffset(7));
325    
326                    // ###
327    
328                    assertTrue(book.isEmpty(BID));
329    
330                    entry = new DefBookEntry(MODIFY, BID, DEFAULT, 1, newPrice(3275, -2),
331                                    newSize(11));
332                    result = book.make(entry);
333                    // System.out.println(book);
334                    assertEquals(result, TOP);
335                    assertEquals(1, bids.placeCount());
336                    assertEquals(1, bids.placeFromOffset(0));
337                    assertEquals(1, bids.placeFromOffset(16 - 1));
338    
339                    entry = new DefBookEntry(MODIFY, BID, DEFAULT, 9, newPrice(3150, -2),
340                                    newSize(13));
341                    result = book.make(entry);
342                    // System.out.println(book);
343                    assertEquals(result, NORMAL);
344                    assertEquals(2, bids.placeCount());
345                    assertEquals(2, bids.placeFromOffset(16 - 16));
346                    assertEquals(2, bids.placeFromOffset(16 - 6));
347                    assertEquals(1, bids.placeFromOffset(16 - 5));
348                    assertEquals(1, bids.placeFromOffset(16 - 1));
349    
350                    entry = new DefBookEntry(MODIFY, BID, DEFAULT, -1, newPrice(3350, -2),
351                                    newSize(15));
352                    result = book.make(entry);
353                    // System.out.println(book);
354                    assertEquals(result, TOP);
355                    assertEquals(3, bids.placeCount());
356                    assertEquals(3, bids.placeFromOffset(16 - 16));
357                    assertEquals(2, bids.placeFromOffset(16 - 8));
358                    assertEquals(1, bids.placeFromOffset(16 - 3));
359                    assertEquals(1, bids.placeFromOffset(16 - 1));
360    
361                    entry = new DefBookEntry(REMOVE, BID, DEFAULT, 1, NULL_PRICE, null);
362                    result = book.make(entry);
363                    // System.out.println(book);
364                    assertEquals(result, TOP);
365                    assertEquals(bids.count(), bids.placeCount());
366                    assertEquals(2, bids.placeCount());
367                    assertEquals(2, bids.placeFromOffset(16 - 16));
368                    assertEquals(1, bids.placeFromOffset(16 - 4));
369                    assertEquals(0, bids.placeFromOffset(16 - 1));
370    
371                    entry = book.topFor(BID);
372                    assertEquals(entry.side(), BID);
373                    assertEquals(entry.type(), COMBO);
374                    assertEquals(entry.place(), 1);
375                    assertEquals(entry.price(), newPrice(3275, -2));
376                    assertEquals(entry.size(), newSize(11));
377    
378            }
