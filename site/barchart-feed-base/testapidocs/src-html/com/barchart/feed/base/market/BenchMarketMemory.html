008    package com.barchart.feed.base.market;
009    
010    import com.barchart.feed.base.instrument.MockDefinitionService;
011    import com.barchart.feed.base.instrument.api.DefinitionService;
012    import com.barchart.feed.base.instrument.values.MarketInstrument;
013    import com.barchart.feed.base.market.api.Market;
014    import com.barchart.feed.base.market.api.MarketTaker;
015    import com.barchart.util.values.api.TextValue;
016    import com.barchart.util.values.provider.ValueBuilder;
017    
018    public class BenchMarketMemory {
019    
020            static MockMaker maker = new MockMaker(new MockMarketFactory());
021    
022            static int COUNT = 100 * 1000;
023    
024            static DefinitionService service = new MockDefinitionService();
025    
026            public static void main(final String[] args) throws Exception {
027    
028                    MarketInstrument[] inst = new MarketInstrument[COUNT];
029    
030                    MarketTaker<Market> taker;
031    
032                    for (int k = 0; k < COUNT; k++) {
033    
034                            final TextValue id = ValueBuilder.newText("market-" + k);
035    
036                            inst = new MarketInstrument[] { service.lookup(id) };
037    
038                            taker = new MockTaker<Market>(inst);
039    
040                            maker.register(taker);
041    
042                    }
043    
044                    System.gc();
045    
046                    Thread.sleep(1000 * 1000);
047    
048            }
049    }
