008    package com.barchart.feed.base.market;
009    
010    import java.util.concurrent.locks.Lock;
011    import java.util.concurrent.locks.ReentrantLock;
012    
013    import org.slf4j.Logger;
014    import org.slf4j.LoggerFactory;
015    
016    import com.barchart.feed.base.provider.DefBookEntry;
017    import com.barchart.feed.base.provider.DefMarket;
018    import com.barchart.feed.base.provider.VarMarket;
019    import com.barchart.util.bench.size.JavaSize;
020    import com.barchart.util.values.api.PriceValue;
021    import com.barchart.util.values.api.SizeValue;
022    import com.barchart.util.values.api.TimeValue;
023    import com.barchart.util.values.provider.ValueBuilder;
024    
025    public class BenchPrimitiveMemory {
026    
027            private static final Logger log = LoggerFactory
028                            .getLogger(BenchPrimitiveMemory.class);
029    
030            static int COUNT = 100 * 1000;
031    
032            static PriceValue[] priceArray = new PriceValue[COUNT];
033            static SizeValue[] sizeArray = new SizeValue[COUNT];
034            static TimeValue[] timeArray = new TimeValue[COUNT];
035            static String[] stringArray = new String[COUNT];
036            static Lock[] lockArray = new ReentrantLock[COUNT];
037            static Object[] objectArray = new Object[COUNT];
038            static DefMarket[] marketArray = new DefMarket[COUNT];
039            static VarMarket[] marketArray2 = new VarMarket[COUNT];
040    
041            static BenchClassByte[] benchByte = new BenchClassByte[COUNT];
042            static BenchClassInt[] benchInt = new BenchClassInt[COUNT];
043            static BenchClassLong[] benchLong = new BenchClassLong[COUNT];
044    
045            static BenchClassByteLong[] benchByteLong = new BenchClassByteLong[COUNT];
046            static BenchClassLongLong[] benchLongLong = new BenchClassLongLong[COUNT];
047    
048            static BenchClassByteLongLong[] benchByteLongLong = new BenchClassByteLongLong[COUNT];
049    
050            static BenchClassByteExt[] benchByteExt = new BenchClassByteExt[COUNT];
051            static BenchClassByteExtExt[] benchByteExtExt = new BenchClassByteExtExt[COUNT];
052    
053            static DefBookEntry[] bookEntryArray = new DefBookEntry[COUNT];
054    
055            public static void main(final String[] args) throws Exception {
056    
057                    log.info("PriceValue {}", JavaSize.of(ValueBuilder.newPrice(0, 0)));
058                    log.info("SizeValue {}", JavaSize.of(ValueBuilder.newSize(0)));
059                    log.info("TimeValue {}", JavaSize.of(ValueBuilder.newTime(0)));
060                    log.info("String {}", JavaSize.of(new String()));
061                    log.info("ReentrantLock {}", JavaSize.of(new ReentrantLock()));
062                    log.info("Object {}", JavaSize.of(new Object()));
063                    log.info("DefMarket {}", JavaSize.of(new DefMarket()));
064                    log.info("VarMarketL1 {}", JavaSize.of(new MockVarMarket()));
065    
066                    log.info("BenchClassByte {}", JavaSize.of(new BenchClassByte()));
067                    log.info("BenchClassInt {}", JavaSize.of(new BenchClassInt()));
068                    log.info("BenchClassLong {}", JavaSize.of(new BenchClassLong()));
069    
070                    log.info("BenchClassByteLong {}", JavaSize.of(new BenchClassByteLong()));
071                    log.info("BenchClassLongLong {}", JavaSize.of(new BenchClassLongLong()));
072                    log.info("BenchClassByteLongLong {}",
073                                    JavaSize.of(new BenchClassByteLongLong()));
074    
075                    log.info("BenchClassByteExt {}", JavaSize.of(new BenchClassByteExt()));
076                    log.info("BenchClassByteExtExt {}",
077                                    JavaSize.of(new BenchClassByteExtExt()));
078    
079                    // log.info("BookEntry {}", JavaSize.of(new BookEntry()));
080    
081                    for (int k = 0; k < COUNT; k++) {
082    
083                            priceArray[k] = ValueBuilder.newPrice(k, 123);
084                            sizeArray[k] = ValueBuilder.newSize(k);
085                            timeArray[k] = ValueBuilder.newTime(k);
086                            stringArray[k] = new String();
087                            lockArray[k] = new ReentrantLock();
088                            objectArray[k] = new Object();
089                            marketArray[k] = new DefMarket();
090                            marketArray2[k] = new MockVarMarket();
091    
092                            benchByte[k] = new BenchClassByte();
093                            benchInt[k] = new BenchClassInt();
094                            benchLong[k] = new BenchClassLong();
095    
096                            benchByteLong[k] = new BenchClassByteLong();
097                            benchLongLong[k] = new BenchClassLongLong();
098    
099                            benchByteLongLong[k] = new BenchClassByteLongLong();
100    
101                            benchByteExt[k] = new BenchClassByteExt();
102                            benchByteExtExt[k] = new BenchClassByteExtExt();
103    
104                            // bookEntryArray[k] = new BookEntry();
105    
106                    }
107    
108                    System.gc();
109    
110                    log.info("READY");
111    
112                    Thread.sleep(1000 * 1000);
113    
114            }
115    
116    }
