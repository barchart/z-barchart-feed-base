057            @Test
058            public void testTakerMarket() {
059    
060                    final MockMaker maker = new MockMaker(new MockMarketFactory());
061    
062                    final MarketInstrument inst;
063    
064                    inst = service.lookup(newText("1"));
065    
066                    final MarketInstrument[] insts = new MarketInstrument[] { inst };
067    
068                    final MarketTaker<Market> tempTaker = new MockTaker<Market>(insts);
069    
070                    maker.register(tempTaker);
071                    assertEquals(maker.marketCount(), 1);
072    
073                    MarketTaker<Market> taker;
074                    boolean isAdded;
075    
076                    final AtomicInteger count = new AtomicInteger(0);
077    
078                    taker = new MarketTaker<Market>() {
079                            @Override
080                            public MarketEvent[] bindEvents() {
081                                    return MarketEvent.values();
082                            }
083    
084                            @Override
085                            public MarketField<Market> bindField() {
086                                    return MarketField.MARKET;
087                            }
088    
089                            @Override
090                            public MarketInstrument[] bindInstruments() {
091                                    return new MarketInstrument[] { inst };
092                            }
093    
094                            Market previous;
095    
096                            @Override
097                            public void onMarketEvent(final MarketEvent event,
098                                            final MarketInstrument instrument, final Market market) {
099    
100                                    count.incrementAndGet();
101    
102                                    log.info("event : {}  market : {}", event, market.hashCode());
103                                    log.info("BAR_CURRENT = " + market.get(BAR_CURRENT).get(VOLUME));
104    
105                                    if (previous == null) {
106                                            previous = market;
107                                    }
108    
109                                    assertTrue(previous == market);
110                                    previous = market;
111    
112                                    assertEquals(inst, instrument);
113    
114                                    final SizeValue volume = market.get(BAR_CURRENT).get(VOLUME);
115    
116                                    assertEquals(volume, newSize(10));
117    
118                            }
119                    };
120                    isAdded = maker.register(taker);
121                    assertTrue(isAdded);
122                    assertEquals(maker.marketCount(), 1);
123    
124                    //
125    
126                    final MockMsgTrade msgTrade = new MockMsgTrade(inst);
127                    msgTrade.type = FUTURE_ELECTRONIC;
128                    msgTrade.session = DEFAULT;
129                    msgTrade.sequencing = NORMAL;
130                    msgTrade.price = newPrice(100, 0);
131                    msgTrade.size = newSize(10);
132                    msgTrade.time = newTime(0);
133                    msgTrade.date = newTime(0);
134    
135                    maker.make(msgTrade);
136    
137                    // This failed after remaking a test version of market, need to debug
138                    // assertEquals(count.get(), 6);
139    
140            }
