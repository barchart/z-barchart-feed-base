053    public class MockMarket extends VarMarket {
054    
055            @Override
056            public void setInstrument(final MarketInstrument symbol) {
057    
058                    final MarketInstrument oldInst = get(INSTRUMENT);
059    
060                    if (InstrumentConst.NULL_INSTRUMENT.equals(oldInst)) {
061                            set(INSTRUMENT, symbol);
062                    } else {
063                            throw new IllegalStateException("symbol can be set only once");
064                    }
065            }
066    
067            @Override
068            public void setBookUpdate(final MarketDoBookEntry entry,
069                            final TimeValue time) {
070    
071                    assert entry != null && time != null;
072    
073                    final MarketDoBook book = (MarketDoBook) get(BOOK);
074    
075                    final UniBookResult result = book.setEntry(entry);
076    
077                    switch (result) {
078                    case TOP:
079                            eventAdd(NEW_BOOK_TOP);
080                            // continue
081                    case NORMAL:
082                            eventAdd(NEW_BOOK_UPDATE);
083                            break;
084                    default:
085                            eventAdd(NEW_BOOK_ERROR);
086                            return;
087                    }
088    
089                    book.setTime(time);
090                    updateMarket(time);
091    
092            }
093    
094            @Override
095            public void setBookSnapshot(final MarketDoBookEntry[] entries,
096                            final TimeValue time) {
097                    assert entries != null;
098                    assert time != null;
099    
100                    final MarketDoBook book = (MarketDoBook) get(BOOK);
101    
102                    book.clear();
103    
104                    for (final MarketDoBookEntry entry : entries) {
105    
106                            if (entry == null) {
107                                    continue;
108                            }
109    
110                            final UniBookResult result = book.setEntry(entry);
111    
112                            switch (result) {
113                            case TOP:
114                            case NORMAL:
115                                    break;
116                            default:
117                                    eventAdd(NEW_BOOK_ERROR);
118                                    break;
119                            }
120    
121                    }
122    
123                    eventAdd(NEW_BOOK_SNAPSHOT);
124    
125                    book.setTime(time);
126                    updateMarket(time);
127    
128            }
129    
130            @Override
131            public void setCuvolUpdate(final MarketDoCuvolEntry entry,
132                            final TimeValue time) {
133    
134                    assert entry != null && time != null;
135    
136                    makeCuvol(entry.price(), entry.size());
137    
138                    updateMarket(time);
139    
140            }
141    
142            @Override
143            public void setCuvolSnapshot(final MarketDoCuvolEntry[] entries,
144                            final TimeValue time) {
145    
146                    assert entries != null && time != null;
147    
148                    final MarketDoCuvol cuvol = loadCuvol();
149    
150                    cuvol.clear();
151    
152                    for (final MarketDoCuvolEntry entry : entries) {
153                            cuvol.add(entry.price(), entry.size());
154                    }
155    
156                    eventAdd(NEW_CUVOL_SNAPSHOT);
157    
158                    updateMarket(time);
159    
160            }
161    
162            @Override
163            public void setBar(final MarketBarType type, final MarketDoBar bar) {
164    
165                    assert type != null;
166                    assert bar != null;
167    
168                    set(type.field, bar);
169    
170                    eventAdd(type.event);
171    
172                    updateMarket(bar.get(BAR_TIME));
173    
174            }
175    
176            @Override
177            public void setTrade(final MarketTradeType type,
178                            final MarketTradeSession session,
179                            final MarketTradeSequencing sequencing, final PriceValue price,
180                            final SizeValue size, final TimeValue time, final TimeValue date) {
181    
182                    assert type != null;
183                    assert price != null;
184                    assert size != null;
185                    assert time != null;
186                    assert date != null;
187    
188                    final MarketDoTrade trade = loadTrade();
189    
190                    trade.set(TYPE, type);
191                    trade.set(PRICE, price);
192                    trade.set(SIZE, size);
193                    trade.set(TRADE_TIME, time);
194                    trade.set(MarketTradeField.TRADE_DATE, date);
195    
196                    applyTradeToBar(CURRENT, price, size, time, date);
197    
198                    makeCuvol(price, size);
199    
200                    updateMarket(time);
201    
202            }
203    
204            private final void updateMarket(final TimeValue time) {
205    
206                    set(MARKET_TIME, time);
207    
208                    eventAdd(MARKET_UPDATED);
209    
210            }
211    
212            private final void makeCuvol(final PriceValue price, final SizeValue size) {
213    
214                    final MarketDoCuvol cuvol = loadCuvol();
215    
216                    cuvol.add(price, size);
217    
218                    eventAdd(NEW_CUVOL_UPDATE);
219    
220            }
221    
222            private final void applyTradeToBar(final MarketBarType type,
223                            final PriceValue price, final SizeValue size, final TimeValue time,
224                            final TimeValue date) {
225    
226                    final MarketDoBar bar = loadBar(type.field);
227    
228                    eventAdd(type.event);
229    
230                    final SizeValue volumeOld = bar.get(VOLUME);
231                    final SizeValue volumeNew = volumeOld.add(size);
232                    bar.set(VOLUME, volumeNew);
233                    eventAdd(NEW_VOLUME);
234    
235                    bar.set(CLOSE, price);
236                    if (type == CURRENT) {
237                            // events only for combo
238                            eventAdd(NEW_CLOSE);
239                    }
240    
241                    // ### time
242    
243                    bar.set(BAR_TIME, time);
244                    bar.set(TRADE_DATE, date);
245    
246            }
247    
248            @Override
249            public void setState(final MarketStateEntry entry, final boolean isOn) {
250    
251                    assert entry != null;
252    
253                    final MarketState state = loadState();
254    
255                    if (isOn) {
256                            state.add(entry);
257                    } else {
258                            state.remove(entry);
259                    }
260    
261            }
