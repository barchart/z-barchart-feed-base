008    package com.barchart.feed.base.provider;
009    
010    import com.barchart.feed.base.market.enums.MarketField;
011    import com.barchart.util.anno.NotMutable;
012    import com.barchart.util.values.api.Value;
013    
014    @NotMutable
015    public class DefMarket extends NulMarket {
016    
017            protected final static int ARRAY_SIZE = MarketField.size();
018    
019            protected final Value<?>[] valueArray;
020    
021            public DefMarket() {
022                    valueArray = new Value<?>[ARRAY_SIZE];
023            }
024    
025            @SuppressWarnings("unchecked")
026            @Override
027            public <V extends Value<V>> V get(final MarketField<V> field) {
028    
029                    assert field != null;
030    
031                    final V value = (V) valueArray[field.ordinal()];
032    
033                    if (value == null) {
034                            return field.value();
035                    } else {
036                            return value;
037                    }
038    
039            }
040    
041    }
