001    package com.barchart.feed.base.cuvol.map.provider;
002    
003    import java.util.Map;
004    
005    import com.barchart.util.collections.PriceArrayMap;
006    import com.barchart.util.values.api.PriceValue;
007    import com.barchart.util.values.api.SizeValue;
008    import com.barchart.util.values.provider.ValueConst;
009    import com.barchart.util.values.util.ValueUtil;
010    
011    public class DefCuvolMap extends NulCuvolMap {
012    
013            public static double EPSILON = 0.0000000001;
014            
015            private final Map<PriceValue, SizeValue> offGridMap;
016            
017            private final SizeValue[] entries;
018            private final PriceArrayMap<SizeValue> map;
019    
020            private final PriceValue priceFirst;
021            private final PriceValue priceStep;
022            
023            DefCuvolMap(final Map<PriceValue, SizeValue> offGridMap, PriceArrayMap<SizeValue> map, 
024                            final PriceValue priceFirst, final PriceValue priceStep) {
025    
026                    assert offGridMap != null;
027                    assert priceFirst != null;
028                    assert priceStep != null;
029                    assert priceStep.mantissa() != 0;
030    
031                    this.offGridMap = offGridMap;
032                    
033                    this.priceFirst = priceFirst;
034                    this.priceStep = priceStep;
035    
036                    this.map = map;
037                    
038                    entries = entries();
039    
040                    for (int k = 0; k < entries.length; k++) {
041                            SizeValue volume = entries[k];
042                            if (volume == null) {
043                                    volume = ValueConst.NULL_SIZE;
044                            }
045                            entries[k] = volume.freeze();
046                    }
047                    
048            }
049            
050            @Override
051            public SizeValue getCuvol(final PriceValue price) {
052                    
053                    SizeValue size = ValueConst.NULL_SIZE;
054                    if(onGrid(price)) {
055                            size = map.get(price);
056                    } else if(offGridMap != null) {
057                            size = offGridMap.get(price);
058                    }
059                    
060                    if(size == null) {
061                            return ValueConst.NULL_SIZE;
062                    } else {
063                            return size;
064                    }
065            }
066            
067            @Override
068            public PriceValue priceFirst() {
069                    return priceFirst;
070            }
071    
072            @Override
073            public PriceValue priceStep() {
074                    return priceStep;
075            }
076    
077            @Override
078            public SizeValue[] entries() {
079                    return entries;
080            }
081            
082            private boolean onGrid(final PriceValue price) {
083                    
084                    //Do without double
085                    final double result = ValueUtil.asDouble(price) / ValueUtil.asDouble(priceStep);
086                    
087                    final long resLong = Math.round(result);
088                    
089                    if(Math.abs(result - resLong) < EPSILON) {
090                            return true;
091                    }
092                    
093                    return false;
094                    
095            }
096            
097    }
