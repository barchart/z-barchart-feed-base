008    package com.barchart.feed.base.provider;
009    
010    import com.barchart.feed.base.trade.api.MarketTrade;
011    import com.barchart.feed.base.trade.enums.MarketTradeField;
012    import com.barchart.util.values.api.Value;
013    import com.barchart.util.values.provider.ValueFreezer;
014    
015    public class NulTrade extends ValueFreezer<MarketTrade> implements MarketTrade {
016    
017            @Override
018            public <V extends Value<V>> V get(final MarketTradeField<V> field) {
019                    assert field != null;
020                    return field.value();
021            }
022    
023            @Override
024            public final boolean isNull() {
025                    return this == MarketConst.NULL_TRADE;
026            }
027    
028            @Override
029            public final String toString() {
030    
031                    final StringBuilder text = new StringBuilder(128);
032    
033                    text.append("Trade > ");
034    
035                    for (final MarketTradeField<?> field : MarketTradeField.values()) {
036                            text.append(get(field));
037                            text.append(" ");
038                    }
039    
040                    // System.err.println(text.length());
041    
042                    return text.toString();
043    
044            }
045    
046    }
