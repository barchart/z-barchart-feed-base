001    package com.barchart.feed.base.trade.enums;
002    
003    import com.barchart.util.values.api.Value;
004    
005    public enum MarketTradeSequencing implements Value<MarketTradeSequencing> {
006    
007            // Null value
008            NULL_TRADE_SEQUENCE,
009    
010            // Regular in-sequence trade
011            NORMAL,
012    
013            // Out-of-sequence trade, may update current high/low but not last
014            UNSEQUENCED,
015    
016            ;
017    
018            @Override
019            public MarketTradeSequencing freeze() {
020                    return this;
021            }
022    
023            @Override
024            public boolean isFrozen() {
025                    return true;
026            }
027    
028            @Override
029            public boolean isNull() {
030                    return this == NULL_TRADE_SEQUENCE;
031            }
032    
033    }
