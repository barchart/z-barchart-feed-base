084            @Override
085            public synchronized final <V extends Value<V>> boolean register(
086                            final MarketTaker<V> taker) {
087    
088                    if (!RegTaker.isValid(taker)) {
089                            return false;
090                    }
091    
092                    RegTaker<?> regTaker = takerMap.get(taker);
093    
094                    final boolean wasAdded = (regTaker == null);
095    
096                    while (regTaker == null) {
097                            regTaker = new RegTaker<V>(taker);
098                            takerMap.putIfAbsent(taker, regTaker);
099                            regTaker = takerMap.get(taker);
100                    }
101    
102                    if (wasAdded) {
103                            for (final MarketInstrument inst : regTaker.getInstruments()) {
104    
105                                    if (!isValid(inst)) {
106                                            continue;
107                                    }
108    
109                                    if (!isRegistered(inst)) {
110                                            register(inst);
111                                    }
112    
113                                    final MarketDo market = marketMap.get(inst);
114    
115                                    market.runSafe(safeRegister, regTaker);
116    
117                                    notifyRegListeners(market);
118    
119                            }
120                    } else {
121                            log.warn("already registered : {}", taker);
122                    }
123    
124                    return wasAdded;
125    
126            }

137            /** TODO optimize for speed */
138            @Override
139            public synchronized final <V extends Value<V>> boolean update(
140                            final MarketTaker<V> taker) {
141    
142                    if (!RegTaker.isValid(taker)) {
143                            // debug logged already
144                            return false;
145                    }
146    
147                    final RegTaker<?> regTaker = takerMap.get(taker);
148    
149                    if (regTaker == null) {
150                            log.warn("Taker not registered : {}", taker);
151                            return false;
152                    }
153    
154                    //
155    
156                    final Set<MarketInstrument> updateSet = new HashSet<MarketInstrument>();
157                    final Set<MarketInstrument> registerSet = new HashSet<MarketInstrument>();
158                    final Set<MarketInstrument> unregisterSet = new HashSet<MarketInstrument>();
159                    final Set<MarketInstrument> changeNotifySet = new HashSet<MarketInstrument>();
160    
161                    {
162    
163                            final MarketInstrument[] pastArray = regTaker.getInstruments();
164                            final MarketInstrument[] nextArray = taker.bindInstruments();
165    
166                            final Set<MarketInstrument> pastSet = new HashSet<MarketInstrument>(
167                                            Arrays.asList(pastArray));
168                            final Set<MarketInstrument> nextSet = new HashSet<MarketInstrument>(
169                                            Arrays.asList(nextArray));
170    
171                            /** past & next */
172                            updateSet.addAll(pastSet);
173                            updateSet.retainAll(nextSet);
174    
175                            /** next - past */
176                            registerSet.addAll(nextSet);
177                            registerSet.removeAll(updateSet);
178    
179                            /** past - next */
180                            unregisterSet.addAll(pastSet);
181                            unregisterSet.removeAll(updateSet);
182    
183                            /** past + next */
184                            changeNotifySet.addAll(updateSet);
185                            changeNotifySet.addAll(registerSet);
186                            changeNotifySet.addAll(unregisterSet);
187    
188                    }
189    
190                    //
191    
192                    // log.debug("updateSet : {}", updateSet);
193                    // log.debug("registerSet : {}", registerSet);
194                    // log.debug("unregisterSet : {}", unregisterSet);
195                    // log.debug("changeNotifySet : {}", changeNotifySet);
196    
197                    //
198    
199                    /** unregister : based on past */
200                    for (final MarketInstrument inst : unregisterSet) {
201    
202                            final MarketDo market = marketMap.get(inst);
203    
204                            market.runSafe(safeUnregister, regTaker);
205    
206                    }
207    
208                    /** update : based on merge of next and past */
209                    for (final MarketInstrument inst : updateSet) {
210    
211                            final MarketDo market = marketMap.get(inst);
212    
213                            market.runSafe(safeUpdate, regTaker);
214    
215                    }
216    
217                    /** past = next */
218                    regTaker.bind();
219    
220                    /** register : based on next */
221                    for (final MarketInstrument inst : registerSet) {
222    
223                            if (!isValid(inst)) {
224                                    continue;
225                            }
226    
227                            if (!isRegistered(inst)) {
228                                    register(inst);
229                            }
230    
231                            final MarketDo market = marketMap.get(inst);
232    
233                            market.runSafe(safeRegister, regTaker);
234    
235                    }
236    
237                    /** remove / notify */
238                    for (final MarketInstrument inst : changeNotifySet) {
239    
240                            final MarketDo market = marketMap.get(inst);
241    
242                            if (!market.hasRegTakers()) {
243                                    unregister(inst);
244                            }
245    
246                            notifyRegListeners(market);
247    
248                    }
249    
250                    return true;
251            }

254            new MarketSafeRunner<Void, RegTaker<?>>() {
255                    @Override

262            @Override
263            public synchronized final <V extends Value<V>> boolean unregister(
264                            final MarketTaker<V> taker) {
265    
266                    if (!RegTaker.isValid(taker)) {
267                            return false;
268                    }
269    
270                    final RegTaker<?> regTaker = takerMap.remove(taker);
271    
272                    final boolean wasRemoved = (regTaker != null);
273    
274                    if (wasRemoved) {
275                            for (final MarketInstrument inst : regTaker.getInstruments()) {
276    
277                                    if (!isValid(inst)) {
278                                            continue;
279                                    }
280    
281                                    final MarketDo market = marketMap.get(inst);
282    
283                                    if(market==null){
284                                            log.error("can get MarketDo for " + inst.get(InstrumentField.ID).toString());
285                                            continue;
286                                    }
287                                    
288                                    market.runSafe(safeUnregister, regTaker);
289    
290                                    if (!market.hasRegTakers()) {
291                                            unregister(inst);
292                                    }
293    
294                                    notifyRegListeners(market);
295    
296                            }
297                    } else {
298                            log.warn("was not registered : {}", taker);
299                    }
300    
301                    return wasRemoved;
302    
303            }

316            @Override
317            public void make(final Message message) {
318    
319                    final MarketInstrument instrument = message.getInstrument();
320    
321                    if (!isValid(instrument)) {
322                            return;
323                    }
324    
325                    final MarketDo market = marketMap.get(instrument);
326    
327                    if (!isValid(market)) {
328                            return;
329                    }
330    
331                    market.runSafe(safeMake, message);
332    
333            }

338                    public Void runSafe(final MarketDo market, final Message message) {
339                            make(message, market);
340                            market.fireEvents();
341                            return null;

349            @SuppressWarnings("unchecked")
350            @Override
351            public final <S extends MarketInstrument, V extends Value<V>> V take(
352                            final S instrument, final MarketField<V> field) {
353    
354                    final MarketDo market = marketMap.get(instrument);
355    
356                    if (market == null) {
357                            return MarketConst.NULL_MARKET.get(field).freeze();
358                    }
359    
360                    return (V) market.runSafe(safeTake, field);
361    
362            }

403            protected boolean isValid(final MarketInstrument instrument) {
404    
405                    if (instrument == null) {
406                            log.error("instrument == null");
407                            return false;
408                    }
409    
410                    if (instrument.isNull()) {
411                            log.error("instrument.isNull()");
412                            return false;
413                    }
414    
415                    final PriceValue priceStep = instrument.get(InstrumentField.PRICE_STEP);
416    
417                    if (priceStep.isZero()) {
418                            log.error("priceStep.isZero()");
419                            return false;
420                    }
421    
422                    final Fraction fraction = instrument.get(InstrumentField.FRACTION);
423    
424                    if (fraction.isNull()) {
425                            log.error("fraction.isNull()");
426                            return false;
427                    }
428    
429                    if (priceStep.exponent() != fraction.decimalExponent) {
430                            log.error("priceStep.exponent() != fraction.decimalExponent");
431                            return false;
432                    }
433    
434                    return true;
435    
436            }

440            @Override
441            public final synchronized boolean register(final MarketInstrument instrument) {
442    
443                    if (!isValid(instrument)) {
444                            return false;
445                    }
446    
447                    MarketDo market = marketMap.get(instrument);
448    
449                    final boolean wasAdded = (market == null);
450    
451                    while (market == null) {
452                            market = factory.newMarket();
453                            market.setInstrument(instrument);
454                            marketMap.putIfAbsent(instrument, market);
455                            market = marketMap.get(instrument);
456                    }
457    
458                    if (wasAdded) {
459                            //
460                    } else {
461                            log.warn("already registered : {}", instrument);
462                    }
463    
464                    return wasAdded;
465    
466            }

502            protected final void notifyRegListeners(final MarketDo market) {
503    
504                    final MarketInstrument inst = market.get(MarketField.INSTRUMENT);
505    
506                    final Set<MarketEvent> events = market.regEvents();
507    
508                    for (final MarketRegListener listener : listenerList) {
509                            try {
510                                    listener.onRegistrationChange(inst, events);
511                            } catch (final Exception e) {
512                                    log.error("", e);
513                            }
514                    }
515    
516            }
517    
518            @Override
519            public void notifyRegListeners() {
520                    for (final MarketDo market : marketMap.values()) {
521                            notifyRegListeners(market);
522                    }
