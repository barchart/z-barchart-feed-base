001    package com.barchart.feed.base.trade.enums;
002    
003    import com.barchart.util.values.api.Value;
004    
005    public enum MarketTradeSession implements Value<MarketTradeSession> {
006    
007            // Null trade session
008            NULL_TRADE_SESSION,
009    
010            // Default (default session for stocks, pit session for futures)
011            DEFAULT,
012    
013            // Pit session
014            PIT,
015    
016            // Combined sessions
017            COMBINED,
018    
019            // Extended trading (Form-T for stocks, overnight for futures)
020            EXTENDED,
021    
022            ;
023    
024            @Override
025            public MarketTradeSession freeze() {
026                    return this;
027            }
028    
029            @Override
030            public boolean isFrozen() {
031                    return true;
032            }
033    
034            @Override
035            public boolean isNull() {
036                    return this == NULL_TRADE_SESSION;
037            }
038    
039    }
