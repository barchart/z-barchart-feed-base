008    package com.barchart.feed.base.book.enums;
009    
010    import com.barchart.util.values.api.Value;
011    
012    public enum MarketBookType implements Value<MarketBookType> {
013    
014            /** no size book */
015            EMPTY, //
016    
017            /** only default sizes */
018            DEFAULT, //
019    
020            /** only implied sizes */
021            IMPLIED, //
022    
023            /** both default + implied sizes */
024            COMBO, //
025    
026            ;
027    
028            public final byte ord = (byte) ordinal();
029    
030            private static final MarketBookType[] ENUM_VALUES = values();
031    
032            public static final MarketBookType fromOrd(final byte ord) {
033                    return ENUM_VALUES[ord];
034            }
035    
036            @Override
037            public MarketBookType freeze() {
038                    return this;
039            }
040    
041            @Override
042            public boolean isFrozen() {
043                    return true;
044            }
045    
046            @Override
047            public boolean isNull() {
048                    return this == EMPTY;
049            }
050    
051    }
