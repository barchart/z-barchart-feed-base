008    package com.barchart.feed.base.provider;
009    
010    import static com.barchart.feed.base.book.enums.MarketBookAction.*;
011    import static com.barchart.feed.base.book.enums.MarketBookSide.*;
012    import static com.barchart.feed.base.book.enums.MarketBookType.*;
013    import static com.barchart.feed.base.provider.MarketConst.*;
014    import static com.barchart.util.values.provider.ValueConst.*;
015    
016    import com.barchart.feed.base.book.api.MarketBookEntry;
017    import com.barchart.feed.base.book.api.MarketDoBookEntry;
018    import com.barchart.feed.base.book.enums.MarketBookAction;
019    import com.barchart.feed.base.book.enums.MarketBookSide;
020    import com.barchart.feed.base.book.enums.MarketBookType;
021    import com.barchart.util.math.MathExtra;
022    import com.barchart.util.values.api.PriceValue;
023    import com.barchart.util.values.api.SizeValue;
024    import com.barchart.util.values.provider.ValueFreezer;
025    
026    // JavaSize this = 8(obj) 4 * 1(byte) + 4(priceRef) + 4(sizeRef) = 24
027    public class DefBookEntry extends ValueFreezer<MarketBookEntry> implements
028                    MarketDoBookEntry {
029    
030            private final static byte nulAct = NOOP.ord;
031            private final static byte nulSide = GAP.ord;
032            private final static byte nulType = EMPTY.ord;
033    
034            // store byte ordinal to save heap
035            private final byte ordAct;
036            private final byte ordSide;
037            private final byte ordType;
038    
039            // place expected to fit byte size
040            private final byte place;
041    
042            // will restore from null
043            private final PriceValue price;
044            private final SizeValue size;
045    
046            public DefBookEntry(final MarketBookAction act, final MarketBookSide side,
047                            final MarketBookType type, final int place, final PriceValue price,
048                            final SizeValue size) throws ArithmeticException {
049    
050                    this.ordAct = (act == null ? nulAct : act.ord);
051                    this.ordSide = (side == null ? nulSide : side.ord);
052                    this.ordType = (type == null ? nulType : type.ord);
053    
054                    this.place = MathExtra.castIntToByte(place);
055    
056                    this.price = price;
057                    this.size = size;
058    
059            }
060    
061            @Override
062            public final MarketBookAction act() {
063                    return MarketBookAction.fromOrd(ordAct);
064            }
065    
066            @Override
067            public final MarketBookSide side() {
068                    return MarketBookSide.fromOrd(ordSide);
069            }
070    
071            @Override
072            public final MarketBookType type() {
073                    return MarketBookType.fromOrd(ordType);
074            }
075    
076            @Override
077            public final int place() {
078                    return place;
079            }
080    
081            @Override
082            public final PriceValue price() {
083                    return (price == null) ? NULL_PRICE : price;
084            }
085    
086            @Override
087            public final SizeValue size() {
088                    return (size == null) ? NULL_SIZE : size;
089            }
090    
091            //
092    
093            @Override
094            public String toString() {
095                    return String.format("%s   %s   %s   %s", side(), place(), price(),
096                                    size());
097            }
098    
099            public String toStringFull() {
100                    return String.format("%s   %s   %s   %s   %s   %s", act(), side(),
101                                    type(), place(), price(), size());
102            }
103    
104            private static final void checkNull(final Object value) {
105                    if (value == null) {
106                            throw new NullPointerException("invalid implementation");
107                    }
108            }
109    
110            static {
111                    final DefBookEntry entry = new DefBookEntry(null, null, null, 0, null,
112                                    null);
113                    checkNull(entry.act());
114                    checkNull(entry.side());
115                    checkNull(entry.type());
116                    checkNull(entry.price());
117                    checkNull(entry.size());
118            }
119    
120            @Override
121            public final int hashCode() {
122                    return ((ordAct << 24) | (ordSide << 16) | (ordType << 8))
123                                    ^ (price().hashCode()) ^ (size().hashCode());
124            }
125    
126            @Override
127            public final boolean equals(final Object thatEntry) {
128                    if (thatEntry instanceof DefBookEntry) {
129                            final DefBookEntry that = (DefBookEntry) thatEntry;
130                            return this.ordAct == that.ordAct && this.ordSide == that.ordSide
131                                            && this.ordType == that.ordType && this.place == that.place
132                                            && this.price().equals(that.price())
133                                            && this.size().equals(that.size());
134                    }
135                    return false;
136            }
137    
138            @Override
139            public final boolean isNull() {
140                    return this == NULL_BOOK_ENTRY;
141            }
142    
143    }
