008    package com.barchart.feed.base.state.enums;
009    
010    import org.slf4j.Logger;
011    import org.slf4j.LoggerFactory;
012    
013    import com.barchart.util.collections.BitSetEnum;
014    
015    public enum MarketStateEntry implements BitSetEnum<MarketStateEntry> {
016    
017            /**
018             * @deprecated IS_SETTLED should be inferred from the values in 
019             * the CURRENT and PREVIOUS bars.
020             */
021            @Deprecated IS_SETTLED,         // Generic Settled Flag
022            IS_PUBLISH_REALTIME,                    // Publishing Realtime, or Other
023            
024            ;
025    
026            private static final Logger log = LoggerFactory
027                            .getLogger(MarketStateEntry.class);
028    
029            private final long mask = ONE << ordinal();
030    
031            @Override
032            public final long mask() {
033                    return mask;
034            }
035    
036            private static final MarketStateEntry[] ENUM_VALUES = values();
037    
038            private static final int ENUM_SIZE = ENUM_VALUES.length;
039    
040            public static final int size() {
041                    return ENUM_SIZE;
042            }
043    
044            static {
045                    log.debug("market state count : {}", ENUM_SIZE);
046            }
047    
048            /** convenience array wrapper */
049            public static MarketStateEntry[] in(final MarketStateEntry... events) {
050                    return events;
051            }
052    
053    }
