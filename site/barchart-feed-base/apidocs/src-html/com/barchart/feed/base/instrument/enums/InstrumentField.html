008    package com.barchart.feed.base.instrument.enums;
009    
010    import static com.barchart.feed.base.instrument.api.InstrumentConst.*;
011    import static com.barchart.feed.base.instrument.enums.MarketDisplay.Fraction.*;
012    import static com.barchart.util.values.provider.ValueConst.*;
013    
014    import com.barchart.feed.base.book.enums.MarketBookType;
015    import com.barchart.feed.base.instrument.enums.MarketDisplay.Fraction;
016    import com.barchart.util.anno.NotMutable;
017    import com.barchart.util.collections.BitSetEnum;
018    import com.barchart.util.enums.DictEnum;
019    import com.barchart.util.enums.ParaEnumBase;
020    import com.barchart.util.values.api.PriceValue;
021    import com.barchart.util.values.api.SizeValue;
022    import com.barchart.util.values.api.TextValue;
023    import com.barchart.util.values.api.TimeValue;
024    import com.barchart.util.values.api.Value;
025    
026    @NotMutable
027    public final class InstrumentField<V extends Value<V>> extends
028                    ParaEnumBase<V, InstrumentField<V>> implements
029                    BitSetEnum<InstrumentField<?>> {
030    
031            // ##################################
032    
033            /** market identifier; must be globally unique; */
034            public static final InstrumentField<TextValue> ID = NEW(NULL_TEXT);
035    
036            /** market security group identifier */
037            public static final InstrumentField<TextValue> GROUP_ID = NEW(NULL_TEXT);
038    
039            /** market originating exchange identifier */
040            public static final InstrumentField<TextValue> EXCHANGE_ID = NEW(NULL_TEXT);
041    
042            /** market symbol; can be non unique; */
043            public static final InstrumentField<TextValue> SYMBOL = NEW(NULL_TEXT);
044    
045            /** market free style description; can be used in full text search */
046            public static final InstrumentField<TextValue> DESCRIPTION = NEW(NULL_TEXT);
047    
048            /** book depth */
049            public static final InstrumentField<SizeValue> BOOK_SIZE = NEW(NULL_SIZE);
050    
051            /** book management type; default vs implied vs combined */
052            public static final InstrumentField<MarketBookType> BOOK_TYPE = NEW(MarketBookType.DEFAULT);
053    
054            /** price step / increment size / tick size */
055            public static final InstrumentField<PriceValue> PRICE_STEP = NEW(NULL_PRICE);
056    
057            /** value of a future contract / stock share */
058            public static final InstrumentField<PriceValue> PRICE_POINT = NEW(NULL_PRICE);
059    
060            /** display fraction render format : decimal vs binary, etc. */
061            public static final InstrumentField<Fraction> FRACTION = NEW(NULL_FRACTION);
062    
063            /** price currency */
064            public static final InstrumentField<MarketCurrency> CURRENCY = NEW(NULL_CURRENCY);
065    
066            /** stock vs future vs etc. */
067            public static final InstrumentField<CodeCFI> TYPE = NEW(CodeCFI.UNKNOWN);
068    
069            // ##################################
070    
071            // /** TODO */
072            // public static final InstrumentField<MarketCalendar> CALENDAR =
073            // NEW(NULL_CALENDAR);
074    
075            /**
076             * market instrument local time zone; can be used to render time in market's
077             * local time;
078             * 
079             * http://joda-time.sourceforge.net/timezones.html
080             */
081            public static final InstrumentField<TextValue> TIME_ZONE = NEW(NULL_TEXT);
082    
083            /** market open time on normal business day */
084            public static final InstrumentField<TimeValue> TIME_OPEN = NEW(NULL_TIME);
085    
086            /** market close time on normal business day */
087            public static final InstrumentField<TimeValue> TIME_CLOSE = NEW(NULL_TIME);
088    
089            /** instrument initiation trading date; such as option issue date */
090            public static final InstrumentField<TimeValue> DATE_START = NEW(NULL_TIME);
091    
092            /** instrument termination trading date; such as future expiration month */
093            public static final InstrumentField<TimeValue> DATE_FINISH = NEW(NULL_TIME);
094    
095            // ##################################
096    
097            public static int size() {
098                    return values().length;
099            }
100    
101            public static InstrumentField<?>[] values() {
102                    return DictEnum.valuesForType(InstrumentField.class);
103            }
104    
105            private final long mask;
106    
107            @Override
108            public long mask() {
109                    return mask;
110            }
111    
112            private InstrumentField() {
113                    super();
114                    mask = 0;
115            }
116    
117            private InstrumentField(final V value) {
118                    super("", value);
119                    mask = ONE << ordinal();
120            }
121    
122            private static final <X extends Value<X>> InstrumentField<X> NEW(X value) {
123                    return new InstrumentField<X>(value);
124            }
125    
126    }
