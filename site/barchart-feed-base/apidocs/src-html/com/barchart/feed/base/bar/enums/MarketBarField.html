008    package com.barchart.feed.base.bar.enums;
009    
010    import static com.barchart.util.values.provider.ValueConst.*;
011    
012    import com.barchart.util.anno.NotMutable;
013    import com.barchart.util.collections.BitSetEnum;
014    import com.barchart.util.enums.DictEnum;
015    import com.barchart.util.enums.ParaEnumBase;
016    import com.barchart.util.values.api.PriceValue;
017    import com.barchart.util.values.api.SizeValue;
018    import com.barchart.util.values.api.TimeValue;
019    import com.barchart.util.values.api.Value;
020    
021    @NotMutable
022    /** represents O-H-L-C market bar fields */
023    public final class MarketBarField<V extends Value<V>> extends
024                    ParaEnumBase<V, MarketBarField<V>> implements
025                    BitSetEnum<MarketBarField<?>> {
026    
027            // ##################################
028    
029            // prices
030    
031            public static final MarketBarField<PriceValue> OPEN = NEW(NULL_PRICE);
032            public static final MarketBarField<PriceValue> HIGH = NEW(NULL_PRICE);
033            public static final MarketBarField<PriceValue> LOW = NEW(NULL_PRICE);
034            public static final MarketBarField<PriceValue> CLOSE = NEW(NULL_PRICE);
035            public static final MarketBarField<PriceValue> SETTLE = NEW(NULL_PRICE);
036    
037            // sizes
038    
039            public static final MarketBarField<SizeValue> VOLUME = NEW(NULL_SIZE);
040            public static final MarketBarField<SizeValue> INTEREST = NEW(NULL_SIZE);
041    
042            /** last time of any bar field change */
043            public static final MarketBarField<TimeValue> BAR_TIME = NEW(NULL_TIME);
044    
045            /**
046             * latest known market trading date represented by this bar
047             * 
048             * expressed as UTC zone
049             * 
050             * year, month, day should be treated as local market trade date
051             */
052            public static final MarketBarField<TimeValue> TRADE_DATE = NEW(NULL_TIME);
053    
054            // ##################################
055    
056            public static int size() {
057                    return values().length;
058            }
059    
060            public static MarketBarField<?>[] values() {
061                    return DictEnum.valuesForType(MarketBarField.class);
062            }
063    
064            private final long mask;
065    
066            @Override
067            public long mask() {
068                    return mask;
069            }
070    
071            private MarketBarField() {
072                    super();
073                    mask = 0;
074            }
075    
076            private MarketBarField(final V value) {
077                    super("", value);
078                    mask = ONE << ordinal();
079            }
080    
081            private static <X extends Value<X>> MarketBarField<X> NEW(
082                            final X defaultValue) {
083                    return new MarketBarField<X>(defaultValue);
084            }
085    
086    }
