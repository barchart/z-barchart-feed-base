004    package com.barchart.feed.base.provider;

006    import com.barchart.feed.base.market.api.MarketDo;
007    import com.barchart.feed.base.market.api.MarketFactory;
008    import com.barchart.feed.base.market.api.MarketProvider;
009    
010    /**
011     * Keeping this for now, but need to revisit. It seems fine to let
012     * MarketMakerProvider just take a MarketFactory, which can be defined in
013     * DDF-feed. If this is sufficient, this can be removed. It is not currently
014     * used anywhere.
015     * 
016     * @author g-litchfield
017     * 
018     */
019    public class MarketProviderImpl implements MarketProvider {
020    
021            @Override
022            public MarketFactory makeFactory(final Class<? extends MarketDo> clazz) {
023    
024                    return new MarketFactory() {
025    
026                            @Override
027                            public MarketDo newMarket() {
028                                    try {
029                                            return clazz.newInstance();
030                                    } catch (final InstantiationException e) {
031                                            e.printStackTrace();
032                                    } catch (final IllegalAccessException e) {
033                                            e.printStackTrace();
034                                    }
035                                    return null;
036                            }
037    
038                    };
039            }
040    }
