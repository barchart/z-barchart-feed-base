008    package com.barchart.feed.base.market.api;
009    
010    import com.barchart.feed.base.instrument.values.MarketInstrument;
011    import com.barchart.feed.base.market.enums.MarketEvent;
012    import com.barchart.feed.base.market.enums.MarketField;
013    import com.barchart.util.anno.EventListener;
014    import com.barchart.util.anno.NotMutable;
015    import com.barchart.util.anno.UsedOnce;
016    import com.barchart.util.values.api.Value;
017    
018    /**
019     * NOTE: bind happens only once at taker registration
020     * 
021     */
022    @NotMutable
023    public interface MarketTaker<V extends Value<V>> {
024    
025            /** Specify value type that will be provided in the {@link #onMarketEvent} */
026            @UsedOnce
027            MarketField<V> bindField();
028    
029            /** Specify list of events this taker will subscribe to */
030            @UsedOnce
031            MarketEvent[] bindEvents();
032    
033            /** Specify list of markets this taker will subscribe to */
034            @UsedOnce
035            MarketInstrument[] bindInstruments();
036    
037            //
038    
039            /**
040             * Event handler method that will receive subscribed events for subscribed
041             * markets; returned values are frozen and disconnected from live market
042             */
043            @EventListener(MarketEvent.class)
044            void onMarketEvent(MarketEvent event, MarketInstrument instrument,
045                            @NotMutable V value);
046    
047            //
048    
049    }
