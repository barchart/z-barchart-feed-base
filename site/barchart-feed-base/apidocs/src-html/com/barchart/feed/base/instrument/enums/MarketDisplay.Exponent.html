149            public static final long priceWhole(/* local */long mantissa, /* local */
150                            int exponent, final Fraction frac) {
151    
152                    final int scale = frac.decimalExponent;
153    
154                    while (exponent > scale) {
155                            mantissa *= 10;
156                            exponent--;
157                    }
158    
159                    while (exponent < scale) {
160                            mantissa /= 10;
161                            exponent++;
162                    }
163    
164                    return mantissa / frac.decimalDenominator;

168            // note: can overflow
169            public static final long priceFraction(/* local */long mantissa, /* local */
170                            int exponent, /* local */Fraction frac) {
171    
172                    if (frac == null) {
173                            frac = Fraction.DEC_Z00;
174                    }
175    
176                    final int scale = frac.decimalExponent;
177    
178                    while (exponent > scale) {
179                            mantissa *= 10;
180                            exponent--;
181                    }
182    
183                    while (exponent < scale) {
184                            mantissa /= 10;
185                            exponent++;
186                    }
187    
188                    mantissa = abs(mantissa);
189    
190                    switch (frac.base) {
191                    default:
192                    case DECIMAL:
193                            mantissa %= frac.denominator;
194                            break;
195                    case BINARY:
196                            mantissa %= frac.decimalDenominator;
197                            mantissa *= frac.denominator;
198                            mantissa /= frac.decimalDenominator;
199                            break;
200                    }
201    
202                    return mantissa;
203    
204            }

223                    }
224

226

258            /**
259             * render price according to BARCHART convention & fraction format; decimal
260             * vs binary; proper number of places;
261             */
262            public static final String priceText(/* local */long mantissa,
263            /* local */int exponent, /* local */Fraction frac) {
264    
265                    if (frac == null) {
266                            frac = Fraction.DEC_Z00;
267                    }
268    
269                    /* consume sign */
270    
271                    final boolean isMinus;
272    
273                    if (mantissa < 0) {
274                            mantissa = -mantissa;
275                            isMinus = true;
276                    } else {
277                            isMinus = false;
278                    }
279    
280                    /* normalize exponent to fraction */
281    
282                    final int scale = frac.decimalExponent;
283    
284                    while (exponent > scale) {
285                            mantissa *= 10;
286                            exponent--;
287                    }
288    
289                    while (exponent < scale) {
290                            mantissa /= 10;
291                            exponent++;
292                    }
293    
294                    /* produce whole */
295    
296                    long whole = mantissa / frac.decimalDenominator;
297    
298                    /* produce part */
299    
300                    final char separator;
301    
302                    switch (frac.base) {
303                    default:
304                    case DECIMAL:
305                            mantissa %= frac.denominator;
306                            separator = ASCII.DOT;
307                            break;
308                    case BINARY:
309                            mantissa %= frac.decimalDenominator;
310                            mantissa *= frac.denominator;
311                            mantissa /= frac.decimalDenominator;
312                            separator = ASCII.DASH;
313                            break;
314                    }
315    
316                    long part = mantissa;
317    
318                    /* render price to array */
319    
320                    final int size = 64;
321    
322                    final char[] array = new char[size];
323    
324                    int index = size - 1;
325    
326                    final int places = frac.places;
327    
328                    /* part */if (places > 0) {
329    
330                            for (int k = 0; k < places; k++) {
331                                    final char alpha = (char) (part % 10 + ASCII._0_);
332                                    array[index--] = alpha;
333                                    part /= 10;
334                            }
335    
336                            array[index--] = separator;
337    
338                    }
339    
340                    /* whole */if (whole == 0) {
341                            array[index--] = ASCII._0_;
342                    } else {
343    
344                            while (whole != 0) {
345                                    final char alpha = (char) (whole % 10 + ASCII._0_);
346                                    array[index--] = alpha;
347                                    whole /= 10;
348                            }
349    
350                            if (isMinus) {
351                                    array[index--] = ASCII.MINUS;
352                            }
353    
354                    }
355    
356                    final int offset = index + 1;
357    
358                    final int length = size - offset;
359    
360                    return new String(array, offset, length);
361    
362            }

523            /** known fraction descriptor */
524            public static enum Fraction implements Value<Fraction> {
525    
526                    NULL_FRACTION(Base.NULL_BASE, Exponent.NULL_EXPONENT,
527                                    "unknown fraction"), //
528    
529                    DEC_Z00(DECIMAL, Z00, "decimal, zero exponent, 1.0"), //
530    
531                    DEC_N01(DECIMAL, N01, "decimal, negative one, 0.1"), //
532                    DEC_N02(DECIMAL, N02, "decimal, negative two, 0.01"), //
533                    DEC_N03(DECIMAL, N03, "decimal, negative three, 0.001"), //
534                    DEC_N04(DECIMAL, N04, "decimal, negative four, 0.000,1"), //
535                    DEC_N05(DECIMAL, N05, "decimal, negative five, 0.000,01"), //
536                    DEC_N06(DECIMAL, N06, "decimal, negative six, 0.000,001"), //
537                    DEC_N07(DECIMAL, N07, "decimal, negative seven, 0.000,000,1"), //
538                    DEC_N08(DECIMAL, N08, "decimal, negative eight, 0.000,000,01"), //
539                    DEC_N09(DECIMAL, N09, "decimal, negative nine, 0.000,000,001"), //
540    
541                    BIN_Z00(BINARY, Z00, "binary, zero exponent, 1/1"), //
542    
543                    BIN_N01(BINARY, N01, "binary, negative one, 1/2"), //
544                    BIN_N02(BINARY, N02, "binary, negative two, 1/4"), //
545                    BIN_N03(BINARY, N03, "binary, negative three, 1/8"), //
546                    BIN_N04(BINARY, N04, "binary, negative four, 1/16"), //
547                    BIN_N05(BINARY, N05, "binary, negative five, 1/32"), //
548                    BIN_N06(BINARY, N06, "binary, negative six, 1/64"), //
549                    BIN_N07(BINARY, N07, "binary, negative seven, 1/128"), //
550                    BIN_N08(BINARY, N08, "binary, negative eight, 1/256"), //
551                    BIN_N09(BINARY, N09, "binary, negative nine, 1/512"), //
552    
553                    ;
554    
555                    public String description;
556    
557                    public final Base base;
558                    public final Exponent exponent;
559    
560                    /** native form, such as 1/10 or 1/32 */
561                    public final long numerator;
562                    public final long denominator;
563    
564                    /** decimal form, such as 1/10 */
565                    public final int decimalExponent;
566                    public final long decimalDenominator;
567    
568                    /** number of PLACES it takes to DISPLAY a fraction in a native form */
569                    public final int places;
570    
571                    Fraction(final Base base, final Exponent exponent,
572                                    final String description) {
573    
574                            this.base = base;
575                            this.exponent = exponent;
576                            this.description = description;
577    
578                            numerator = 1;
579                            denominator = (long) pow(base.value, -exponent.value);
580    
581                            decimalExponent = exponent.value;
582                            decimalDenominator = (long) pow(10, -decimalExponent);
583    
584                            switch (base) {
585                            case BINARY:
586                                    if (denominator == 1) {
587                                            places = 0;
588                                    } else {
589                                            places = (int) (1 + log10(denominator));
590                                    }
591                                    break;
592                            default:
593                            case DECIMAL:
594                                    places = -exponent.value;
595                                    break;
596                            }
597    
598                    }
599    
600                    public final boolean isSmallerThan(final Fraction that) {
601                            if (that == null) {
602                                    return false;
603                            }
604                            return this.decimalExponent < that.decimalExponent;
605                    }
606    
607                    private static final Fraction[] ENUM_VALUES = values();
