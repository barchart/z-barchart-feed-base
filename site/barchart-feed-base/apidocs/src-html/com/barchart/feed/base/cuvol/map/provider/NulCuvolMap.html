008    package com.barchart.feed.base.cuvol.map.provider;
009    
010    import com.barchart.feed.base.cuvol.map.api.MarketCuvolMap;
011    import com.barchart.feed.base.provider.MarketConst;
012    import com.barchart.util.anno.NotMutable;
013    import com.barchart.util.values.api.PriceValue;
014    import com.barchart.util.values.api.SizeValue;
015    import com.barchart.util.values.provider.ValueConst;
016    import com.barchart.util.values.provider.ValueFreezer;
017    
018    @NotMutable
019    public class NulCuvolMap extends ValueFreezer<MarketCuvolMap> implements MarketCuvolMap {
020    
021            @Override
022            public SizeValue getCuvol(PriceValue price) {
023                    return ValueConst.NULL_SIZE;
024            }
025    
026            @Override
027            public PriceValue priceFirst() {
028                    return ValueConst.NULL_PRICE;
029            }
030    
031            @Override
032            public PriceValue priceStep() {
033                    return ValueConst.NULL_PRICE;
034            }
035    
036            @Override
037            public SizeValue[] entries() {
038                    return ValueConst.NULL_SIZE_ARRAY;
039            }
040    
041            @Override
042            public final String toString() {
043    
044                    final StringBuilder text = new StringBuilder(512);
045    
046                    text.append("Cumulative Volume");
047                    text.append("\n");
048                    text.append("Price Step :  ");
049                    text.append(priceStep());
050                    text.append("\n");
051                    text.append("Price First : ");
052                    text.append(priceFirst());
053                    text.append("\n");
054    
055                    for (final SizeValue entry : entries()) {
056                            text.append(entry);
057                            text.append("\n");
058                    }
059    
060                    return text.toString();
061    
062            }
063            
064            @Override
065            public final boolean isNull() {
066                    return this == MarketConst.NULL_CUVOL_MAP;
067            }
068            
069    }
