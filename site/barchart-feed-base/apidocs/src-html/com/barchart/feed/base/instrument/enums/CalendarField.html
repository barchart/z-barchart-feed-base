008    package com.barchart.feed.base.instrument.enums;
009    
010    import static com.barchart.util.values.provider.ValueConst.*;
011    
012    import com.barchart.util.anno.NotMutable;
013    import com.barchart.util.collections.BitSetEnum;
014    import com.barchart.util.enums.DictEnum;
015    import com.barchart.util.enums.ParaEnumBase;
016    import com.barchart.util.values.api.TextValue;
017    import com.barchart.util.values.api.TimeValue;
018    import com.barchart.util.values.api.Value;
019    
020    /**
021     * represents instrument time events;
022     * 
023     * all times are in UTC;
024     */
025    @NotMutable
026    public final class CalendarField<V extends Value<V>> extends
027                    ParaEnumBase<V, CalendarField<V>> implements
028                    BitSetEnum<CalendarField<?>> {
029    
030            // ##################################
031    
032            /**
033             * market instrument local time zone; can be used to render time in market's
034             * local time;
035             * 
036             * http://joda-time.sourceforge.net/timezones.html
037             */
038            public static final CalendarField<TextValue> TIME_ZONE = NEW(NULL_TEXT);
039    
040            /** market open time on normal business day */
041            public static final CalendarField<TimeValue> TIME_OPEN = NEW(NULL_TIME);
042    
043            /** market close time on normal business day */
044            public static final CalendarField<TimeValue> TIME_CLOSE = NEW(NULL_TIME);
045    
046            /** instrument initiation trading date; such as option issue date */
047            public static final CalendarField<TimeValue> DATE_START = NEW(NULL_TIME);
048    
049            /** instrument termination trading date; such as future expiration month */
050            public static final CalendarField<TimeValue> DATE_FINISH = NEW(NULL_TIME);
051    
052            // ##################################
053    
054            public static int size() {
055                    return values().length;
056            }
057    
058            public static CalendarField<?>[] values() {
059                    return DictEnum.valuesForType(CalendarField.class);
060            }
061    
062            private final long mask;
063    
064            @Override
065            public long mask() {
066                    return mask;
067            }
068    
069            private CalendarField() {
070                    super();
071                    mask = 0;
072            }
073    
074            private CalendarField(final V value) {
075                    super("", value);
076                    mask = ONE << ordinal();
077            }
078    
079            private static final <X extends Value<X>> CalendarField<X> NEW(X value) {
080                    return new CalendarField<X>(value);
081            }
082    
083    }
