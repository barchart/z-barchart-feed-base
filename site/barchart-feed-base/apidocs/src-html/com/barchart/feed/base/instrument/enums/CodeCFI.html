031            /* keep most specific first; 'X' means 'any' */
032

036            EQUITY_OTHER("EMXXXX"), //

046            FUTURE_FINANCIAL("FFXXXX"), //
047    
048            FUTURE("FXXXXX"), //
049    
050            /* option */
051    
052            OPTION_CALL_EUROPEAN("OCEXXX"), //
053            OPTION_PUT_EUROPEAN("OPEXXX"), //
054            OPTION_CALL_AMERICAN("OCAXXX"), //
055            OPTION_PUT_AMERICAN("OPAXXX"), //
056    
057            OPTION_SPREAD("OMXXXX"), //
058            OPTION_CALL("OCXXXX"), //
059            OPTION_PUT("OPXXXX"), //

061            OPTION("OXXXXX"), //

064

079            UNKNOWN("??????"), //

082    
083            private static final Logger log = LoggerFactory.getLogger(CodeCFI.class);
084    
085            public static final char ANY = 'X';
086            public static final char NONE = '?';
087            public static final int SIZE = 6;
088    
089            /**  */
090            private final String code;
091    
092            /** full text search description */
093            private final String description;
094    
095            private CodeCFI(final String code) {
096    
097                    assert code.length() == SIZE;
098    
099                    this.code = code;
100    
101                    this.description = name().replaceAll("_", " ");
102    
103            }
104    
105            public String getCode() {
106                    return code;
107            }
108    
109            public String getDescription() {
110                    return description;
111            }
112    
113            private static final CodeCFI[] ENUM_VALS = values();
114    
115            private static CharSequence padAny(final CharSequence codeCFI) {
116    
117                    final char[] array = new char[SIZE];
118    
119                    final int sizeCFI = codeCFI.length();
120    
121                    for (int index = 0; index < SIZE; index++) {
122    
123                            final char alpha;
124                            if (index < sizeCFI) {
125                                    alpha = codeCFI.charAt(index);
126                            } else {
127                                    alpha = ANY;
128                            }
129    
130                            array[index] = alpha;
131    
132                    }
133    
134                    return new String(array);
135            }
136    
137            /** resolve most specific designation */
138            public static CodeCFI fromCode(/* local */CharSequence codeCFI) {
139    
140                    if (codeCFI == null) {
141                            return UNKNOWN;
142                    }
143    
144                    if (codeCFI.length() < SIZE) {
145                            codeCFI = padAny(codeCFI);
146                    }
147    
148                    for (final CodeCFI known : ENUM_VALS) {
149                            if (known.equalsMask(codeCFI)) {
150                                    return known;
151                            }
152                    }
153    
154                    log.debug("unknown codeCFI={}", codeCFI);
155    
156                    return UNKNOWN;
157    
158            }
159    
160            private boolean equalsMask(final CharSequence codeCFI) {
161    
162                    charLoop: for (int index = 0; index < SIZE; index++) {
163    
164                            final char thisChar = this.code.charAt(index);
165    
166                            if (thisChar == ANY) {
167                                    // ignore
168                                    continue charLoop;
169                            }
170    
171                            if (thisChar == codeCFI.charAt(index)) {
172                                    // match
173                                    continue charLoop;
174                            }
175    
176                            return false; // first non match
177    
178                    }
179    
180                    return true; // total match
181    
182            }
183

188                    return this;
189            }
190

192            public boolean isFrozen() {
193                    return true;
194            }
195    
196            @Override
197            public boolean isNull() {
198                    return this == UNKNOWN;
199            }
