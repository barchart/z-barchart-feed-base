001    package com.barchart.feed.base.trade.enums;
002    
003    import static com.barchart.feed.base.trade.enums.MarketTradeSequencing.NORMAL;
004    import static com.barchart.feed.base.trade.enums.MarketTradeSequencing.UNSEQUENCED;
005    import static com.barchart.feed.base.trade.enums.MarketTradeSession.COMBINED;
006    import static com.barchart.feed.base.trade.enums.MarketTradeSession.DEFAULT;
007    import static com.barchart.feed.base.trade.enums.MarketTradeSession.EXTENDED;
008    import static com.barchart.feed.base.trade.enums.MarketTradeSession.PIT;
009    
010    import com.barchart.util.values.api.Value;
011    
012    public enum MarketTradeType implements Value<MarketTradeType> {
013    
014            // Null trade type
015            NULL_TRADE_TYPE(DEFAULT, NORMAL),
016    
017            // Unknown trade type
018            UNKNOWN(DEFAULT, NORMAL),
019    
020            /** FUTURES */
021    
022            // Composite session (pit + electronic)
023            FUTURE_COMPOSITE(COMBINED, NORMAL),
024            // Electronic/overnight session
025            FUTURE_ELECTRONIC(DEFAULT, NORMAL),
026            // Pit / open outcry session
027            FUTURE_PIT(PIT, NORMAL),
028    
029            /** EQUITIES */
030    
031            // Regular trade types
032            REGULAR_SALE(DEFAULT, NORMAL),
033            ACQUISITION(DEFAULT, NORMAL),
034            AMEX_RULE_155(DEFAULT, NORMAL),
035            AUTOMATIC_EXECUTION(DEFAULT, NORMAL),
036            BUNCHED_TRADE(DEFAULT, NORMAL),
037            CROSS_TRADE(DEFAULT, NORMAL),
038            DISTRIBUTION(DEFAULT, NORMAL),
039            INTERMARKET_SWEEP(DEFAULT, NORMAL),
040            MARKET_CLOSING(DEFAULT, NORMAL),
041            MARKET_OPENING(DEFAULT, UNSEQUENCED),
042            MARKET_REOPENING(DEFAULT, NORMAL),
043            NYSE_RULE_127(DEFAULT, NORMAL),
044            RESERVED(DEFAULT, NORMAL),
045            SPLIT(DEFAULT, NORMAL),
046            STOCK_OPTION(DEFAULT, NORMAL),
047            STOPPED_STOCK_REGULAR(DEFAULT, NORMAL),
048            YELLOW_FLAG(DEFAULT, NORMAL),
049    
050            // Regular out-of-sequence trade types
051            BUNCHED_SOLD(DEFAULT, UNSEQUENCED),
052            DERIVATIVELY_PRICED(DEFAULT, UNSEQUENCED),
053            PRIOR_REFERENCE_PRICE(DEFAULT, UNSEQUENCED),
054            SOLD_LAST(DEFAULT, UNSEQUENCED),
055            SOLD_OOO(DEFAULT, UNSEQUENCED),
056            STOPPED_STOCK_OOO(DEFAULT, UNSEQUENCED),
057            STOPPED_STOCK_SOLD_LAST(DEFAULT, UNSEQUENCED),
058    
059            // Form-T (pre/post-market) trades
060            FORM_T(EXTENDED, NORMAL),
061            FORM_T_OOO(EXTENDED, UNSEQUENCED),
062    
063            ;
064    
065            public final byte ord;
066            public final MarketTradeSession session;
067            public final MarketTradeSequencing sequencing;
068    
069            private final static MarketTradeType[] ENUM_VALUES = values();
070            private final static int ENUM_SIZE = ENUM_VALUES.length;
071    
072            public static final int size() {
073                    return ENUM_SIZE;
074            }
075    
076            private MarketTradeType(final MarketTradeSession se,
077                            final MarketTradeSequencing sq) {
078                    ord = (byte) ordinal();
079                    session = se;
080                    sequencing = sq;
081            }
082    
083            @Override
084            public MarketTradeType freeze() {
085                    return this;
086            }
087    
088            @Override
089            public boolean isFrozen() {
090                    return true;
091            }
092    
093            @Override
094            public boolean isNull() {
095                    return this == NULL_TRADE_TYPE;
096            }
097    
098    }
