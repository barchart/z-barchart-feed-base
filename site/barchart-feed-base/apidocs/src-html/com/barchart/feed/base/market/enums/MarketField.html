011    import static com.barchart.feed.base.provider.MarketConst.*;

013    
014    import com.barchart.feed.base.bar.api.MarketBar;
015    import com.barchart.feed.base.book.api.MarketBook;
016    import com.barchart.feed.base.book.api.MarketBookEntry;
017    import com.barchart.feed.base.book.api.MarketBookTop;
018    import com.barchart.feed.base.cuvol.api.MarketCuvol;
019    import com.barchart.feed.base.cuvol.api.MarketCuvolEntry;

023    import com.barchart.feed.base.trade.api.MarketTrade;

026    import com.barchart.util.enums.DictEnum;
027    import com.barchart.util.enums.ParaEnumBase;
028    import com.barchart.util.values.api.TimeValue;
029    import com.barchart.util.values.api.Value;
030    
031    @NotMutable
032    public final class MarketField<V extends Value<V>> extends
033                    ParaEnumBase<V, MarketField<V>> implements BitSetEnum<MarketField<?>> {
034    
035            // ##################################
036    
037            /** primary key */
038            public static final MarketField<MarketInstrument> INSTRUMENT = NEW(NULL_INSTRUMENT);
039    
040            /** last time any change of any market field */
041            public static final MarketField<TimeValue> MARKET_TIME = NEW(NULL_TIME);
042    
043            /** last trade, any source */
044            public static final MarketField<MarketTrade> TRADE = NEW(NULL_TRADE);
045    
046            /** market depth, a.k.a book */
047            public static final MarketField<MarketBook> BOOK = NEW(NULL_BOOK);
048    
049            /** proxy to last updated entry */
050            public static final MarketField<MarketBookEntry> BOOK_LAST = NEW(NULL_BOOK_ENTRY);
051    
052            /** proxy to market depth top bid & top ask */
053            public static final MarketField<MarketBookTop> BOOK_TOP = NEW(NULL_BOOK_TOP);
054    
055            /** cumulative volume - price / size ladder */
056            public static final MarketField<MarketCuvol> CUVOL = NEW(NULL_CUVOL);
057    
058            /** proxy to last updated entry in cumulative volume */
059            public static final MarketField<MarketCuvolEntry> CUVOL_LAST = NEW(NULL_CUVOL_ENTRY);
060    
061            // market bars aka sessions
062    
063            /** current or default or combo = electronic + manual */
064            public static final MarketField<MarketBar> BAR_CURRENT = NEW(NULL_BAR);
065    
066            /** extra / stocks form t */
067            public static final MarketField<MarketBar> BAR_CURRENT_EXT = NEW(NULL_BAR);
068    
069            /** past - previous day; default or combo */
070            public static final MarketField<MarketBar> BAR_PREVIOUS = NEW(NULL_BAR);
071    
072            // MarketState
073            public static final MarketField<MarketState> STATE = NEW(NULL_STATE);
074    
075            //
076    
077            /** market container with all market fields (self reference) */
078            // NOTE: keep market last
079            public static final MarketField<Market> MARKET = NEW(NULL_MARKET);
080    
081            // ##################################
082    
083            public static int size() {
084                    return values().length;
085            }
086    
087            public static MarketField<?>[] values() {
088                    return DictEnum.valuesForType(MarketField.class);
089            }
090    
091            //
092    
093            private final long mask;
094    
095            @Override
096            public long mask() {
097                    return mask;
098            }
099    
100            private MarketField() {
101                    super();
102                    mask = 0;
103            }
104    
105            private MarketField(final V value) {
106                    super("", value);
107                    mask = ONE << ordinal();
108            }
109    
110            private static final <X extends Value<X>> MarketField<X> NEW(final X value) {
111                    return new MarketField<X>(value);
112            }
113    
114    }
